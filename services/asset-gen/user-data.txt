#!/bin/bash
if ! grep -q "ECS_CLUSTER" /etc/ecs/ecs.config; then
    echo ECS_CLUSTER=gameforge-gpu-cluster >> /etc/ecs/ecs.config
fi
if ! grep -q "ECS_ENABLE_GPU_SUPPORT" /etc/ecs/ecs.config; then
    echo ECS_ENABLE_GPU_SUPPORT=true >> /etc/ecs/ecs.config
fi
systemctl restart ecs
mkdir -p /tmp/gpu_cache /tmp/model_cache
chmod 777 /tmp/gpu_cache /tmp/model_cache
